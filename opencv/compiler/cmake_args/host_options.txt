# Host platform (macOS/Linux) CMake options for OpenCV

# Enable comprehensive feature set for desktop platforms
-DBUILD_SHARED_LIBS=ON

# Enable all image format support
-DBUILD_JPEG=ON
-DBUILD_PNG=ON
-DBUILD_TIFF=ON
-DBUILD_WEBP=ON
-DBUILD_OPENEXR=ON
-DBUILD_JASPER=ON
-DBUILD_OPENJPEG=ON

# Enable video and camera support
-DWITH_FFMPEG=ON
-DWITH_GSTREAMER=ON

# Platform-specific video support
-DWITH_AVFOUNDATION=ON  # macOS
-DWITH_GTK=ON           # Linux
-DWITH_V4L=ON           # Linux

# Enable parallel processing
-DWITH_TBB=ON
-DWITH_OPENMP=ON

# Enable hardware acceleration
-DWITH_OPENCL=ON
-DWITH_OPENCL_SVM=ON
-DWITH_IPP=ON

# Enable all OpenCV modules
-DBUILD_opencv_world=ON

# Enable development features
-DBUILD_TESTS=ON
-DBUILD_PERF_TESTS=ON
-DBUILD_EXAMPLES=ON
-DBUILD_DOCS=ON

# Enable language bindings
-DBUILD_opencv_python3=ON
-DBUILD_opencv_java=ON

# Enable additional features
-DWITH_PROTOBUF=ON
-DWITH_EIGEN=ON
-DWITH_QUIRC=ON
